```python
import scrapy
from scrapy import Spider
from scrapy.http import Response
from items import SpotItem, ReviewItem
import hashlib
from datetime import datetime

class MafengwoSpotSpider(Spider):
    """马蜂窝景点爬虫"""

    name = "mafengwo_spot"
    allowed_domains = ["mafengwo.cn"]
    custom_settings = {
        'DOWNLOAD_DELAY': 2,
        'RANDOMIZE_DOWNLOAD_DELAY': True,
        'CONCURRENT_REQUESTS': 2,
        'CONCURRENT_REQUESTS_PER_DOMAIN': 1,
    }

    def __init__(self, city_id=None, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.city_id = city_id

    def start_requests(self):
        """生成初始请求"""
        # 从城市列表开始
        if self.city_id:
            url = f"https://www.mafengwo.cn/poi/{self.city_id}.html"
            yield scrapy.Request(url, callback=self.parse_city)
        else:
            # 获取城市列表
            url = "https://www.mafengwo.cn/mdd/"
            yield scrapy.Request(url, callback=self.parse_city_list)

    def parse_city_list(self, response):
        """解析城市列表"""
        cities = response.css('.city-list a::attr(href)').getall()
        for city_url in cities:
            city_id = city_url.split('/')[-1].replace('.html', '')
            yield scrapy.Request(
                f"https://www.mafengwo.cn/poi/{city_id}.html",
                callback=self.parse_city,
                meta={'city_id': city_id}
            )

    def parse_city(self, response):
        """解析城市景点列表"""
        city_id = response.meta.get('city_id')

        # 获取景点列表
        spots = response.css('.scenic-list .item')
        for spot in spots:
            spot_url = spot.css('a::attr(href)').get()
            if spot_url:
                yield scrapy.Request(
                    response.urljoin(spot_url),
                    callback=self.parse_spot,
                    meta={'city_id': city_id}
                )

        # 翻页
        next_page = response.css('.pg-next::attr(href)').get()
        if next_page:
            yield scrapy.Request(
                response.urljoin(next_page),
                callback=self.parse_city,
                meta={'city_id': city_id}
            )

    def parse_spot(self, response):
        """解析景点详情"""
        city_id = response.meta.get('city_id')

        # 基础信息
        item = SpotItem()
        item['source'] = 'mafengwo'
        item['source_id'] = response.url.split('/')[-1].replace('.html', '')
        item['source_url'] = response.url
        item['name'] = response.css('.title::text').get('').strip()
        item['name_en'] = response.css('.en-name::text').get('')

        # 位置信息
        item['city'] = response.css('.location .city::text').get('')
        item['address'] = response.css('.location .address::text').get('')

        # 坐标 (从页面JS中提取)
        lat = response.css('meta[property="og:latitude"]::attr(content)').get()
        lng = response.css('meta[property="og:longitude"]::attr(content)').get()
        if lat and lng:
            item['latitude'] = float(lat)
            item['longitude'] = float(lng)

        # 描述
        item['description'] = ''.join(
            response.css('.summary p::text').getall()
        ).strip()

        # 评分
        rating_text = response.css('.rating .score::text').get()
        if rating_text:
            item['rating'] = float(rating_text)

        # 标签
        item['tags'] = response.css('.tags a::text').getall()

        # 图片
        images = response.css('.gallery img::attr(src)').getall()
        item['images'] = [img.replace('_s.', '_l.') for img in images]

        # 开放时间
        item['opening_hours'] = response.css('.open-time::text').get('')

        # 门票
        item['ticket_price'] = response.css('.ticket .price::text').get('')

        # 生成唯一ID
        item['unique_id'] = self._generate_id(item)

        yield item

        # 爬取评价
        review_url = f"https://www.mafengwo.cn/poi/{item['source_id']}/comment.html"
        yield scrapy.Request(
            review_url,
            callback=self.parse_reviews,
            meta={'spot_id': item['unique_id']}
        )

    def parse_reviews(self, response):
        """解析评价"""
        spot_id = response.meta.get('spot_id')

        reviews = response.css('.comment-item')
        for review in reviews:
            item = ReviewItem()
            item['spot_id'] = spot_id
            item['source'] = 'mafengwo'
            item['author'] = review.css('.author::text').get('')
            item['rating'] = len(review.css('.star.active'))
            item['content'] = review.css('.content::text').get('').strip()
            item['date'] = review.css('.date::text').get('')
            item['images'] = review.css('.photos img::attr(src)').getall()

            yield item

    def _generate_id(self, item: SpotItem) -> str:
        """生成唯一ID"""
        key = f"{item['source']}:{item['source_id']}"
        return hashlib.md5(key.encode()).hexdigest()


class MafengwoPipeline:
    """马蜂窝数据处理管道"""

    def process_item(self, item, spider):
        """处理数据项"""
        if isinstance(item, SpotItem):
            return self._process_spot(item)
        elif isinstance(item, ReviewItem):
            return self._process_review(item)
        return item

    def _process_spot(self, item: SpotItem) -> SpotItem:
        """处理景点数据"""
        # 数据清洗
        if item.get('name'):
            item['name'] = item['name'].strip()

        if item.get('description'):
            item['description'] = item['description'].strip()

        # 标签标准化
        if item.get('tags'):
            item['tags'] = [
                self._normalize_tag(tag)
                for tag in item['tags']
                if tag.strip()
            ]

        # 分类推断
        item['category'] = self._infer_category(item)

        return item

    def _normalize_tag(self, tag: str) -> str:
        """标签标准化"""
        tag_map = {
            '自然风光': ['自然', '山水', '风景', '山川'],
            '人文古迹': ['古迹', '历史', '古建筑', '寺庙'],
            '美食': ['美食', '小吃', '餐厅'],
            '摄影': ['摄影', '拍照', '打卡'],
        }

        tag_lower = tag.lower()
        for standard, variants in tag_map.items():
            if tag_lower in variants or any(v in tag_lower for v in variants):
                return standard

        return tag

    def _infer_category(self, item: SpotItem) -> str:
        """推断景点分类"""
        tags = item.get('tags', [])
        name = item.get('name', '')

        # 基于标签推断
        if any(t in tags for t in ['山', '湖', '海', '森林', '峡谷']):
            return 'nature'
        if any(t in tags for t in ['古镇', '寺庙', '博物馆', '古迹']):
            return 'culture'

        # 基于名称推断
        nature_keywords = ['山', '湖', '河', '谷', '林', '海', '岛']
        if any(k in name for k in nature_keywords):
            return 'nature'

        return 'other'
```

```python
class AntiScrapingMiddleware:
    """反爬虫中间件"""

    def __init__(self):
        self.proxy_pool = ProxyPool()
        self.user_agent_pool = UserAgentPool()
        self.cookie_manager = CookieManager()

    def process_request(self, request, spider):
        """处理请求"""
        # 1. 随机User-Agent
        request.headers['User-Agent'] = self.user_agent_pool.get_random()

        # 2. 代理IP
        if self._should_use_proxy(request):
            proxy = self.proxy_pool.get()
            request.meta['proxy'] = proxy

        # 3. Cookie管理
        domain = request.url.split('/')[2]
        if self.cookie_manager.has_cookies(domain):
            request.headers['Cookie'] = self.cookie_manager.get_cookies(domain)

        # 4. 请求指纹
        request.headers['X-Request-ID'] = str(uuid.uuid4())

        return None

    def process_response(self, request, response, spider):
        """处理响应"""
        # 检测被封
        if response.status == 403 or '验证码' in response.text:
            spider.logger.warning(f"Blocked: {request.url}")
            self._handle_block(request, response)
            return request.replace(dont_filter=True)

        # 更新Cookie
        if 'Set-Cookie' in response.headers:
            domain = request.url.split('/')[2]
            self.cookie_manager.update(domain, response.headers['Set-Cookie'])

        return response

    def _should_use_proxy(self, request) -> bool:
        """判断是否使用代理"""
        # 重要页面使用代理
        important_patterns = ['/poi/', '/jd/', '/scenery/']
        return any(p in request.url for p in important_patterns)
```

```python
import overpy
from typing import List, Dict

class OSMDataCollector:
    """OpenStreetMap数据采集"""

    def __init__(self):
        self.api = overpy.Overpass()

    def collect_tourism_pois(self, bbox: tuple) -> List[Dict]:
        """
        采集旅游POI

        bbox: (south, west, north, east)
        """
        query = f"""
        [out:json][timeout:60];
        (
            node["tourism"~"attraction|museum|theme_park|zoo"]{bbox};
            way["tourism"~"attraction|museum|theme_park|zoo"]{bbox};
            node["historic"]{bbox};
            way["historic"]{bbox};
            node["natural"~"peak|beach|waterfall|cave_entrance"]{bbox};
            way["natural"~"peak|beach|waterfall|cave_entrance"]{bbox};
        );
        out center;
        """

        result = self.api.query(query)

        pois = []
        for node in result.nodes:
            poi = self._parse_node(node)
            if poi:
                pois.append(poi)

        for way in result.ways:
            poi = self._parse_way(way)
            if poi:
                pois.append(poi)

        return pois

    def _parse_node(self, node) -> Dict:
        """解析节点"""
        tags = node.tags

        return {
            'osm_id': node.id,
            'name': tags.get('name', ''),
            'name_en': tags.get('name:en', ''),
            'latitude': float(node.lat),
            'longitude': float(node.lon),
            'category': self._map_category(tags),
            'tags': self._extract_tags(tags),
            'description': tags.get('description', ''),
            'website': tags.get('website', ''),
            'opening_hours': tags.get('opening_hours', ''),
        }

    def _map_category(self, tags: Dict) -> str:
        """映射分类"""
        if 'tourism' in tags:
            tourism_map = {
                'attraction': 'attraction',
                'museum': 'museum',
                'theme_park': 'theme_park',
                'zoo': 'zoo',
            }
            return tourism_map.get(tags['tourism'], 'other')

        if 'historic' in tags:
            return 'heritage'

        if 'natural' in tags:
            return 'nature'

        return 'other'

    def _extract_tags(self, tags: Dict) -> List[str]:
        """提取标签"""
        tag_keywords = [
            'natural', 'historic', 'tourism', 'amenity',
            'leisure', 'sport', 'cuisine'
        ]

        result = []
        for key in tag_keywords:
            if key in tags:
                result.append(tags[key])

        return result
```

```python
import requests
from bs4 import BeautifulSoup

class WikivoyageCollector:
    """Wikivoyage数据采集"""

    BASE_URL = "https://zh.wikivoyage.org/wiki/"

    def collect_city_guides(self, city_names: List[str]) -> List[Dict]:
        """采集城市旅行指南"""
        guides = []

        for city in city_names:
            try:
                guide = self._fetch_city_guide(city)
                if guide:
                    guides.append(guide)
                time.sleep(1)  # 礼貌爬取
            except Exception as e:
                self.logger.error(f"Error fetching {city}: {e}")

        return guides

    def _fetch_city_guide(self, city: str) -> Dict:
        """获取单个城市指南"""
        url = f"{self.BASE_URL}{city}"
        response = requests.get(url)

        if response.status_code != 200:
            return None

        soup = BeautifulSoup(response.text, 'html.parser')

        guide = {
            'city': city,
            'source': 'wikivoyage',
            'source_url': url,
            'sections': {}
        }

        # 解析各部分
        sections = soup.find_all(['h2', 'h3'])
        current_section = None

        for heading in sections:
            section_name = heading.get_text().replace('[编辑]', '').strip()
            content = []

            # 获取该部分内容
            sibling = heading.find_next_sibling()
            while sibling and sibling.name not in ['h2', 'h3']:
                if sibling.name == 'p':
                    content.append(sibling.get_text())
                sibling = sibling.find_next_sibling()

            if content:
                guide['sections'][section_name] = '\n'.join(content)

        # 提取景点
        guide['attractions'] = self._extract_attractions(soup)

        return guide

    def _extract_attractions(self, soup) -> List[Dict]:
        """提取景点信息"""
        attractions = []

        # 查找景点列表
        lists = soup.find_all('ul')
        for ul in lists:
            items = ul.find_all('li')
            for item in items:
                # 检查是否是景点
                link = item.find('a')
                if link and 'title' in link.attrs:
                    attraction = {
                        'name': link.get_text(),
                        'description': item.get_text(),
                    }
                    attractions.append(attraction)

        return attractions
```

### 2.4 数据去重与合并

```python
class SpotDeduplicator:
    """景点去重与合并"""

    def __init__(self):
        self.name_similarity_threshold = 0.85
        self.location_threshold = 100  # 米

    def deduplicate(self, spots: List[Spot]) -> List[Spot]:
        """去重"""
        unique_spots = []
        seen = set()

        for spot in spots:
            # 生成指纹
            fingerprint = self._generate_fingerprint(spot)

            if fingerprint in seen:
                continue

            # 检查相似景点
            similar = self._find_similar(spot, unique_spots)

            if similar:
                # 合并数据
                merged = self._merge_spots(similar, spot)
                unique_spots.remove(similar)
                unique_spots.append(merged)
            else:
                unique_spots.append(spot)
                seen.add(fingerprint)

        return unique_spots

    def _generate_fingerprint(self, spot: Spot) -> str:
        """生成指纹"""
        key = f"{spot.source}:{spot.source_id}"
        return hashlib.md5(key.encode()).hexdigest()

    def _find_similar(self, spot: Spot, candidates: List[Spot]) -> Optional[Spot]:
        """查找相似景点"""
        for candidate in candidates:
            # 位置相似
            distance = haversine_distance(
                spot.latitude, spot.longitude,
                candidate.latitude, candidate.longitude
            )

            if distance < self.location_threshold:
                # 名称相似
                name_sim = self._name_similarity(spot.name, candidate.name)
                if name_sim > self.name_similarity_threshold:
                    return candidate

        return None

    def _name_similarity(self, name1: str, name2: str) -> float:
        """计算名称相似度"""
        # 编辑距离
        from difflib import SequenceMatcher
        return SequenceMatcher(None, name1, name2).ratio()

    def _merge_spots(self, primary: Spot, secondary: Spot) -> Spot:
        """合并景点数据"""
        merged = Spot(**primary.dict())

        # 合并标签
        merged.tags = list(set(primary.tags + secondary.tags))

        # 合并图片
        merged.images = list(set(primary.images + secondary.images))

        # 选择更完整的描述
        if len(secondary.description) > len(primary.description):
            merged.description = secondary.description

        # 合并评价
        merged.review_count = primary.review_count + secondary.review_count
        merged.rating = (
            primary.rating * primary.review_count +
            secondary.rating * secondary.review_count
        ) / merged.review_count if merged.review_count > 0 else 0

        # 记录数据来源
        merged.sources = list(set(primary.sources + secondary.sources))

        return merged
```

```python
class EnrichEventFn(DoFn):
    """事件增强"""

    def process(self, element):
        event = element

        # 添加时间维度
        event['date'] = event['timestamp'][:10]
        event['hour'] = event['timestamp'][11:13]
        event['day_of_week'] = datetime.fromisoformat(event['timestamp']).weekday()
```

```python
        elif temp < 10 or temp > 35:

        elif 15 <= temp <= 25:
            score *= 1.0
        else:
```

```python
class TextModerator:
    """文本审核"""

    def __init__(self):
        self.sensitive_words = self._load_sensitive_words()
        self.api_client = ContentSafetyAPIClient()

    def moderate(self, text: str) -> ModerationResult:
        """审核文本"""
        # 1. 敏感词检测
        if self._contains_sensitive(text):
            return ModerationResult(
                status='rejected',
                reason='包含敏感词'
            )

        # 2. API审核
        api_result = self.api_client.analyze(text)

        if api_result['severity'] == 'high':
            return ModerationResult(
                status='rejected',
                reason=api_result['reason']
            )
        elif api_result['severity'] == 'medium':
            return ModerationResult(
                status='review_required',
                reason=api_result['reason']
            )

        return ModerationResult(status='approved')

    def _contains_sensitive(self, text: str) -> bool:
        """检测敏感词"""
        text_lower = text.lower()
        return any(word in text_lower for word in self.sensitive_words)
```

```python
        level_up = self._check_level_up(user_id)

        return RewardResult(
            success=True,
            points=points,
            total_points=self._get_total_points(user_id),
```

```python
    def check_spot_data(self, spot: Spot) -> QualityReport:

        required_fields = ['name', 'latitude', 'longitude', 'country']
        for field in required_fields:
            if not getattr(spot, field, None):
                issues.append(f"缺少必填字段: {field}")
                score -= 20

        if spot.name:

                issues.append("名称过短")
                score -= 10
            if spot.name.isdigit():
                issues.append("名称不能纯数字")
                score -= 20

        # 坐标有效性
        if spot.latitude and spot.longitude:
            if not (-90 <= spot.latitude <= 90):
                issues.append("纬度无效")
                score -= 20
            if not (-180 <= spot.longitude <= 180):
                issues.append("经度无效")
                score -= 20

        # 描述质量
        if spot.description:
            if len(spot.description) < 20:
                issues.append("描述过短")
                score -= 5
        else:
            issues.append("缺少描述")
            score -= 10

        # 图片检查
        if not spot.images:
            issues.append("缺少图片")
            score -= 10

        # 标签检查
        if not spot.tags:
            issues.append("缺少标签")
            score -= 5

        return QualityReport(

            score=max(0, score),
            issues=issues,
```
